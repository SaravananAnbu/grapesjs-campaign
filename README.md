# GrapesJS Campaign

## Download

* `npm i grapesjs-campaign`

# Usage

```js
import gjs-campaign from 'grapesjs-campaign';

var editor = grapesjs.init({
      fromElement: 1,
      container : '#gjs',
      plugins: ['gjs-campaign'],
      pluginsOpts: {
        'gjs-campaign': {/* ...options */}
      }
})
